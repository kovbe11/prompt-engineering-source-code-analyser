    HashSet<Integer> intersect(List<Integer> a, List<Integer> b) {
        var result = new HashSet<Integer>();
        for (int i = 0; i < a.size(); i++) {
            if(b.contains(a.get(i))) {
                result.add(a.get(i));
            }
        }
        for (int i = 0; i < b.size(); i++) {
            if(a.contains(b.get(i))) {
                result.add(b.get(i));
            }
        }
        return result;
    }
